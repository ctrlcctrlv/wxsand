#wxSand: Owen Piette's Falling Sand Game
#Version 4.4, File subversion 2
group fruit lemon grapes
group cheats brewery x
element empty 0 0 0 0 0 0 0 1
#put 1 tiny drop in water to make lemonade
element lemon 255 255 0 1 0 2 0 1
neighbor 1 lemon water 1 lemonade lemonade
#put 1 tiny drop in water to meke wine
element grapes 255 0 255 1 0 2 0 1
neighbor 1 grapes water 1 wine wine
group liquids water
#water 
element water 0 0 255 1 1 0 0 1
group cups glass
#good for making a cup
element glass 255 255 255 0 0 10 0 1
element lemonade 255 200 0 1 1 0 0 0
neighbor 1 lemonade water 1 lemonade lemonade
element wine 255 0 100 1 0 0 0 0
neighbor 1 wine water 1 wine wine
neighbor 1 wine fire 1 fire fire
element brewery 200 0 0 0 0 0 0 1
neighbor 1 brewery empty 1 brewery wine
group vegandherbs tealeaves pot fire
#put in pot and ad water or put water in first cuz it looks cooler
element tealeaves 0 255 0 1 1 3 0 1
neighbor 1 tealeaves water 1 coldtea coldtea
neighbor 1 tealeaves tea 1 tea tea
neighbor 1 tealeaves coldtea 1 tea tea 
neighbor 1 coldtea hotpot 1 tea tea
element coldtea 0 255 50 1 1 0 0 0
#heat it to boil your tea
element pot 255 255 199 0 0 10 0 1
self    0.001     pot   1.0     hotpot 
hotself 50     pot
neighbor 1 pot fire 1 hotpot hotpot
neighbor 1 pot hotpot 1 hotpot hotpot
element hotpot 255 0 0 0 0 10 1 0
element tea 0 255 25 1 0 2 0 0
neighbor 1 tea water 1 tea tea
#use to light x or a pot
element fire 255 100 0 -1 2 0 2 1
#like a stove but weird
element x 0 0 200 0 0 10 1 1
neighbor 1 x fire 1 hotx hotx
element hotx 255 0 233 0 0 0 0 0
neighbor 1 hotx empty 1 hotx fire 
group solids wall
#a wall no falling sand game is complete without it
element wall 255 255 240 0 0 50 0 1
