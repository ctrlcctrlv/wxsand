Add a line of Cloud to the top of the screen and ground to the bottom. Then add water. Pollution for Acid Rain, or Charger for Lightning! 

#wxSand: Owen Piette's Falling Sand Game
#Version 4, File subversion 2
#element element r g b gravity slip density conductivity visible
group Empty Empty
group Start Ground Sky
group Water_Cycle Water
group Acid_Rain Pollution Purifier
group Lightning Charger De-Charger
group Credits iwillcontrolyou dicey111

#-------Background-------#
#Erases.
element	Empty 0 0 0 0 1 0 0 1

#-------Water Cycle Elements-------#
#Ground/Earth.
element Ground 102 51 0 0 1 1 0 1

#Starting Water.
element	Water 32 32 255 0.7 1 0.5 0 1

#Water.
element Ground_Water 32 32 255 .7 1 0.5 0 0

#From Ground to Sky.
element Water_Particle 153 255 255 -1 0.5 0 0 0

#Sky/Atmosphere.
element Sky 254 254 254 0 1 0 0 1

#Cloud.
element Cloud 255 255 255 -0.3 0.5 0 0 0

#Rain Water.
element	Rain_Water 32 32 255 0.5 1 0.5 0 0

#-------Acid Rain Elements-------#
#-------Pollution Elements-------#
#Pollutes Ground.
element Pollution 51 51 51 .7 .5 .9 0 1
self .01 Pollution 1 Empty

#Polluted Ground
element PGround 51 0 0 0 1 1 0 0

#Polluted Water.
element PGround_Water 0 255 255 .7 1 0.5 0 0

#Polluted Water Particle.
element PWater_Particle 0 204 255 -1 0.5 0 0 0

#Polluted Sky.
element PSky 152 254 203 0 1 0 0 0

#Polluted Cloud.
element PCloud 153 255 204 -0.3 0.5 0 0 0

#Polluted Rain/Acid.
element PRain_Water 0 255 0 0.7 1 0.7 0 0

#-------Purifier Elements-------#
#Purifies Ground, Purifing all parts of the water cycle.
element Purifier 255 204 255 .7 .5 .9 0 1
self .01 Purifier 1 Empty

#Purified Ground.
element CGround 102 51 0 0 1 1 0 0
self .01 CGround 1 Ground

#Purified Water.
element CGround_Water 32 32 255 .7 1 0.5 0 0
self .01 CGround_Water 1 Ground_Water

#Purified Water Particle.
element CWater_Particle 153 255 255 -1 0.5 0 0 0
self .001 CWater_Particle 1 Water_Particle

#Purified Cloud.
element CCloud 255 255 255 -0.3 0.5 0 0 0
self .01 CCloud 1 Cloud

#Purified Sky.
element CSky 254 254 254 0 1 0 0 0
self .01 CSky 1 Sky

#-------Lightning Elements-------#
#Positively Charged Particle. Starts the Formation of Lightning.
element Charger 255 69 0 .7 .5 .9 0 1
self .01 Charger 1 Empty

#Negatively Charged Particle. Changes lightning back to water cycle.
element De-Charger 0 69 255 .7 .5 .9 0 1
self .01 De-Charger 1 Empty

#Charged Sky.
element ChSky 255 200 200 0 1 0 0 0

#Lightning.
